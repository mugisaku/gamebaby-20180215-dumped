square_wave  sq;
noise        no;


routine
main()
{
  sq.frequency=440;
  sq.volume=0;
  sq.keyon_flag=1;

    while(sq.volume < 30)
    {
        if(sq.volume < 10)
        {
          print 1;
        }

      else
        if(sq.volume < 20)
        {
          print 2;
        }

      else
        {
          print 3;
        }


      sq.volume += 1;

      sleep 200;
    }


  print 4;
}
